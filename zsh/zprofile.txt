# Homebrew-Pfade automatisch setzen (empfohlen von Homebrew)
eval "$(/opt/homebrew/bin/brew shellenv)"

# Locale (LC_ALL nicht global setzen)
export LANG=de_DE.UTF-8
export LC_CTYPE=de_DE.UTF-8

# Homebrew Cask-Zielverzeichnis setzen
export HOMEBREW_CASK_OPTS="--appdir=/Applications"

# Homebrew: keine Analytics
export HOMEBREW_NO_ANALYTICS=1

# Eigenes Skriptverzeichnis in PATH aufnehmen
export PATH="$PATH:$HOME/Skripte"

# Editor/Pager
export EDITOR="nano"
export VISUAL="$EDITOR"
export PAGER="less"
export LESS="-R"

# Keine Less-History-Datei speichern
export LESSHISTFILE=-

# GPG-Programme korrekt an TTY binden
export GPG_TTY=$(tty)
